# Create structure for categories table CREATE TABLE categories ( id int(11) AUTO_INCREMENT, name varchar(150) NOT NULL, description text DEFAULT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP(), PRIMARY KEY (id) ); CREATE TABLE products ( id int(11) AUTO_INCREMENT, category_id int(11) DEFAULT NULL, name varchar(150) NOT NULL, price int(11) DEFAULT 0, description text DEFAULT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP(), PRIMARY KEY (id), FOREIGN KEY (category_id) REFERENCES categories(id) );
